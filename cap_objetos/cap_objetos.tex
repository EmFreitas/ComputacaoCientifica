#include <assert.h>

//Array de estruturas.


class estrutura{
	public:
		double x,y;
		int n;};



int main(int argn, char** argc){
	estrutura conteudo[2]; 
	
	conteudo[0].x=10;
	conteudo[0].y=10;
	conteudo[0].n=3;
	
	conteudo[1].x=20;
	conteudo[1].y=20;
	conteudo[1].n=6;

	std::cout <<"conteudo[1].x = " << conteudo[1].x << std::endl;

	//Mas qual a diferença entre estrutura e classe?
	return 0;
}
